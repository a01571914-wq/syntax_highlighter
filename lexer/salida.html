// Base class
class Animal {
protected
    string name;

public
    Animal(string n) {
        name = n;
    }

    virtual void speak() {
        cout << Animal sound << endl;
    }
};

// Derived class
class Dog  public Animal {

public
    Dog(string n)  Animal(n) {}
